# 6. 결과값 정리
# 데이터 개수
n <- length(x)
# 구간별 누적도수
xcf <- cumsum(xh$counts)
# 구간별 상대도수
xrf <- xh$counts/n
round(xrf, 4)
##  [1] 0.0267 0.0467 0.0867 0.1533 0.2400 0.1600 0.1200 0.0667 0.0600 0.0200
## [11] 0.0133 0.0067
# 구간별 상대누적도수
xrcf <- xcf/n
round(xrcf, 4)
##  [1] 0.0267 0.0733 0.1600 0.3133 0.5533 0.7133 0.8333 0.9000 0.9600 0.9800
## [11] 0.9933 1.0000
# 구간을 문자열 백터로 정리
xclass <- paste0("(", xh$breaks[-13], ", ", xh$breaks[-1], "]"); xclass

## [11] "(4, 4.2]"   "(4.2, 4.4]"
# 7. 표 형식으로 한 번에 출력
# 열별로 묶음
xtab <- cbind(xh$mids, xh$counts, xcf, xrf, xrcf)
# 행과 열의 이름 지정
rownames(xtab) <- xclass
colnames(xtab) <- c("대푯값", "두수", "누적도수", "상대도수", "상대누적도수")
# 화면에 출력
print(xtab)
